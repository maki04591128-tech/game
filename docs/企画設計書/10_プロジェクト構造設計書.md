## 5. Engine層設計

### 5.1 ゲームループ

```csharp
public class GameLoop
{
    private readonly ITurnManager _turnManager;
    private readonly IInputHandler _inputHandler;
    private readonly IRenderer _renderer;
    private readonly IEventSystem _eventSystem;

    private GameState _state;
    private bool _isRunning;

    public async Task RunAsync(CancellationToken cancellationToken)
    {
        _isRunning = true;

        while (_isRunning && !cancellationToken.IsCancellationRequested)
        {
            // 1. 入力取得（プレイヤーターンの場合）
            if (_turnManager.IsPlayerTurn)
            {
                var input = await _inputHandler.GetInputAsync(cancellationToken);
                ProcessPlayerInput(input);
            }

            // 2. ターン処理
            _turnManager.ProcessTurn(_state);

            // 3. イベント処理
            _eventSystem.ProcessPendingEvents();

            // 4. 状態更新
            UpdateGameState();

            // 5. 描画
            _renderer.Render(_state);

            // 6. 終了判定
            CheckGameEnd();
        }
    }
}
```

### 5.2 ターンシステム

```csharp
public class TurnManager : ITurnManager
{
    private readonly PriorityQueue<ITurnActor, int> _turnQueue;
    private long _currentTurn;

    public bool IsPlayerTurn => CurrentActor is Player;
    public ITurnActor CurrentActor { get; private set; }
    public long CurrentTurn => _currentTurn;

    public void ProcessTurn(GameState state)
    {
        // 行動順に基づいてアクターを取得
        CurrentActor = _turnQueue.Dequeue();

        // アクションを決定・実行
        var action = CurrentActor.DecideAction(state);
        CurrentActor.ExecuteAction(action, state);

        // ターン消費に基づいて再キューイング
        var nextPriority = CalculateNextPriority(CurrentActor, action.TurnCost);
        _turnQueue.Enqueue(CurrentActor, nextPriority);

        // グローバルターンカウント更新
        _currentTurn += action.TurnCost;

        // 時間経過イベント発火
        OnTurnAdvanced?.Invoke(this, new TurnEventArgs(_currentTurn));
    }
}
```

        // 部屋の生成

        // 部屋の配置

        PlaceItems(map, parameters.ItemDensity);

        // トラップの配置

    }

### 5.4 戦闘システム

```csharp
public class CombatSystem : ICombatSystem
{
    private readonly IDamageCalculator _damageCalculator;
    private readonly IHitCalculator _hitCalculator;
    private readonly IEventSystem _eventSystem;

    public CombatResult ExecuteAttack(Character attacker, Character target, AttackType attackType)
    {
        // 命中判定
        var hitResult = _hitCalculator.Calculate(attacker, target, attackType);
        if (!hitResult.IsHit)
        {
            return CombatResult.Miss(attacker, target);
        }

        // ダメージ計算
        var damage = _damageCalculator.Calculate(attacker, target, attackType, hitResult.IsCritical);

        // ダメージ適用
        target.TakeDamage(damage);

        // イベント発火
        _eventSystem.Raise(new CombatEvent(attacker, target, damage));

        // 死亡判定
        if (!target.IsAlive)
        {
            HandleDeath(target, attacker);
        }

        return CombatResult.Hit(attacker, target, damage);
    }
}
```

public class SaveManager : ISaveManager
{
    private const string SaveDirectory = "data/saves";

    public async Task SaveGameAsync(GameState state, string slotName)
    {
        var saveData = new SaveData
        {
            Version = SaveData.CurrentVersion,
            Timestamp = DateTime.UtcNow,
            Player = SerializePlayer(state.Player),
            World = SerializeWorld(state.World),
            Knowledge = SerializeKnowledge(state.Player.Knowledge),
            Flags = SerializeFlags(state.Player.Flags),
            CurrentTurn = state.CurrentTurn
        };

        var json = JsonSerializer.Serialize(saveData, _jsonOptions);
        var path = Path.Combine(SaveDirectory, $"{slotName}.json");

        await File.WriteAllTextAsync(path, json);
    }

    public async Task<GameState> LoadGameAsync(string slotName)
    {
        var path = Path.Combine(SaveDirectory, $"{slotName}.json");
        var json = await File.ReadAllTextAsync(path);

        var saveData = JsonSerializer.Deserialize<SaveData>(json, _jsonOptions);

        // バージョン移行が必要な場合
        if (saveData.Version < SaveData.CurrentVersion)
        {
            saveData = MigrateSaveData(saveData);
        }

        return DeserializeGameState(saveData);
    }
}

### 6.2 死に戻りシステム

```csharp
public class RebirthManager : IRebirthManager
{
    private readonly ISaveManager _saveManager;
    private readonly IKnowledgeTransfer _knowledgeTransfer;

    public async Task<GameState> ExecuteRebirthAsync(Player player, DeathCause cause)
    {
        // 正気度減少量を計算
        int sanityLoss = CalculateSanityLoss(cause);
        int newSanity = Math.Max(0, player.Sanity - sanityLoss);

        // 引き継ぎデータを抽出
        TransferData transferData;

        if (newSanity == 0)
        {
            // 正気度0: フラグのみ保持
            transferData = new TransferData
            {
                Flags = player.Flags.Clone(),
                Achievements = player.Achievements.Clone(),
                // スキル、魔法言語、宗教等は全て消失
                Skills = new SkillSet(),
                MagicLanguage = new MagicLanguage(),
                Religion = null,
                Knowledge = new PlayerKnowledge()
            };
        }
        else
        {
            // 通常の死に戻り: 知識系を引き継ぎ
            transferData = _knowledgeTransfer.Extract(player);
        }

        // 新しいゲーム状態を生成
        var newState = CreateNewGameState(player.Background, transferData, newSanity);

        // 救済カウントの更新（正気度0の場合）
        if (newSanity == 0)
        {
            newState.RescueCount--;
            if (newState.RescueCount < 0)
            {
                // 真のゲームオーバー
                await _saveManager.DeleteSaveAsync(player.SaveSlot);
                throw new TrueGameOverException();
            }
            // 救済による正気度回復
            newState.Player.SetSanity(20);
        }

        return newState;
    }
}
```
