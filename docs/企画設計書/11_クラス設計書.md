```csharp
namespace RougelikeGame.Core.Entities;

/// <summary>
/// 全てのキャラクター（プレイヤー、敵、NPC）の基底クラス
/// </summary>
public abstract class Character : IEntity, ITurnActor, IDamageable
{
    #region Identity
    public Guid Id { get; init; } = Guid.NewGuid();
    public string Name { get; set; } = string.Empty;
    public string Description { get; set; } = string.Empty;
    #endregion

    #region Position
    public Position Position { get; set; }
    public Direction FacingDirection { get; set; } = Direction.South;
    #endregion

    #region Stats
    public Stats BaseStats { get; protected set; }
    public Stats EffectiveStats => CalculateEffectiveStats();

    protected virtual Stats CalculateEffectiveStats()
    {
        var stats = BaseStats;
        foreach (var modifier in GetAllStatModifiers())
        {
            stats = stats.Apply(modifier);
        }
        return stats;
    }

    protected virtual IEnumerable<StatModifier> GetAllStatModifiers()
    {
        // 装備、バフ、デバフなどからの修正を集約
        return StatusEffects
            .Where(e => e.StatModifier.HasValue)
            .Select(e => e.StatModifier!.Value);
    }
    #endregion

    #region Resources
    private int _currentHp;
    public int CurrentHp 
    { 
        get => _currentHp;
        protected set => _currentHp = Math.Clamp(value, 0, MaxHp);
    }
    public int MaxHp => EffectiveStats.MaxHp;

    private int _currentMp;
    public int CurrentMp
    {
        get => _currentMp;
        protected set => _currentMp = Math.Clamp(value, 0, MaxMp);
    }
    public int MaxMp => EffectiveStats.MaxMp;

    private int _currentSp;
    public int CurrentSp
    {
        get => _currentSp;
        protected set => _currentSp = Math.Clamp(value, 0, MaxSp);
    }
    public int MaxSp => EffectiveStats.MaxSp;
    #endregion

    #region Status
    public bool IsAlive => CurrentHp > 0;
    public Faction Faction { get; set; } = Faction.Neutral;
    public List<StatusEffect> StatusEffects { get; } = new();
    #endregion

    #region Combat
    public virtual void TakeDamage(Damage damage)
    {
        int finalDamage = CalculateFinalDamage(damage);
        CurrentHp -= finalDamage;

        OnDamaged?.Invoke(this, new DamageEventArgs(damage, finalDamage));

        if (!IsAlive)
        {
            OnDeath?.Invoke(this, new DeathEventArgs(damage));
        }
    }

    protected virtual int CalculateFinalDamage(Damage damage)
    {
        // 属性耐性、防御力などを考慮
        int baseDamage = damage.Amount;

        // 防御計算
        int defense = GetDefenseAgainst(damage.Type);
        int reducedDamage = Math.Max(1, baseDamage - defense);

        // 属性耐性
        float resistance = GetResistanceAgainst(damage.Element);
        int finalDamage = (int)(reducedDamage * (1f - resistance));

        return Math.Max(1, finalDamage);
    }

    protected virtual int GetDefenseAgainst(DamageType type) => type switch
    {
        DamageType.Physical => EffectiveStats.PhysicalDefense,
        DamageType.Magical => EffectiveStats.MagicalDefense,
        _ => 0
    };

    protected virtual float GetResistanceAgainst(Element element) => 0f;

    public virtual void Heal(int amount)
    {
        CurrentHp += amount;
        OnHealed?.Invoke(this, new HealEventArgs(amount));
    }

    public virtual void RestoreMp(int amount) => CurrentMp += amount;
    public virtual void RestoreSp(int amount) => CurrentSp += amount;
    #endregion

    #region Status Effects
    public void ApplyStatusEffect(StatusEffect effect)
    {
        // 既存の同タイプ効果をチェック
        var existing = StatusEffects.FirstOrDefault(e => e.Type == effect.Type);
        if (existing != null)
        {
            // スタック可能なら重複、そうでなければ上書き
            if (effect.IsStackable)
            {
                existing.Stack(effect);
            }
            else if (effect.Priority >= existing.Priority)
            {
                StatusEffects.Remove(existing);
                StatusEffects.Add(effect);
            }
        }
        else
        {
            StatusEffects.Add(effect);
        }

        OnStatusEffectApplied?.Invoke(this, new StatusEffectEventArgs(effect));
    }

    public void RemoveStatusEffect(StatusEffectType type)
    {
        var effect = StatusEffects.FirstOrDefault(e => e.Type == type);
        if (effect != null)
        {
            StatusEffects.Remove(effect);
            OnStatusEffectRemoved?.Invoke(this, new StatusEffectEventArgs(effect));
        }
    }

    public void TickStatusEffects()
    {
        var expiredEffects = new List<StatusEffect>();

        foreach (var effect in StatusEffects)
        {
            effect.Tick(this);
            if (effect.IsExpired)
            {
                expiredEffects.Add(effect);
            }
        }

        foreach (var effect in expiredEffects)
        {
            StatusEffects.Remove(effect);
            OnStatusEffectRemoved?.Invoke(this, new StatusEffectEventArgs(effect));
        }
    }
    #endregion

    #region Turn System
    public virtual int ActionPriority => EffectiveStats.Agility;

    public abstract TurnAction DecideAction(GameState state);
    public abstract void ExecuteAction(TurnAction action, GameState state);
    #endregion

    #region Events
    public event EventHandler<DamageEventArgs>? OnDamaged;
    public event EventHandler<HealEventArgs>? OnHealed;
    public event EventHandler<DeathEventArgs>? OnDeath;
    public event EventHandler<StatusEffectEventArgs>? OnStatusEffectApplied;
    public event EventHandler<StatusEffectEventArgs>? OnStatusEffectRemoved;
    #endregion
}
```

```csharp
namespace RougelikeGame.Core.Entities;

/// <summary>
/// プレイヤーキャラクター
/// </summary>
public class Player : Character
{
    #region Character Creation
    public Race Race { get; init; }
    public CharacterClass Class { get; init; }
    public Background Background { get; init; }
    #endregion

    #region Progression
    private int _level = 1;
    public int Level 
    { 
        get => _level;
        private set => _level = Math.Clamp(value, 1, MaxLevel);
    }
    public const int MaxLevel = 50;

    public int Experience { get; private set; }
    public int ExperienceToNextLevel => CalculateExpRequired(Level);

    private static int CalculateExpRequired(int level)
    {
        // 経験値曲線: 100 * 1.15^(level-1)
        return (int)(100 * Math.Pow(1.15, level - 1));
    }

    public void GainExperience(int amount)
    {
        Experience += amount;
        while (Experience >= ExperienceToNextLevel && Level < MaxLevel)
        {
            Experience -= ExperienceToNextLevel;
            LevelUp();
        }
    }

    private void LevelUp()
    {
        Level++;
        // ステータス上昇はクラスと種族に依存
        var statGrowth = Class.GetStatGrowth() + Race.GetStatGrowth();
        BaseStats = BaseStats.Apply(statGrowth);

        // HP/MP全回復
        CurrentHp = MaxHp;
        CurrentMp = MaxMp;

        OnLevelUp?.Invoke(this, new LevelUpEventArgs(Level, statGrowth));
    }
    #endregion

    #region Sanity (正気度)
    private int _sanity = 100;
    public int Sanity
    {
        get => _sanity;
        private set => _sanity = Math.Clamp(value, 0, MaxSanity);
    }
    public const int MaxSanity = 100;

    public SanityStage SanityStage => Sanity switch
    {
        >= 80 => SanityStage.Normal,
        >= 60 => SanityStage.Uneasy,
        >= 40 => SanityStage.Anxious,
        >= 20 => SanityStage.Unstable,
        > 0 => SanityStage.Madness,
        _ => SanityStage.Broken
    };

    public void DecreaseSanity(int amount)
    {
        var oldStage = SanityStage;
        Sanity -= CalculateSanityReduction(amount);
        var newStage = SanityStage;

        if (newStage != oldStage)
        {
            OnSanityStageChanged?.Invoke(this, new SanityStageEventArgs(oldStage, newStage));
        }
    }

    private int CalculateSanityReduction(int baseAmount)
    {
        float modifier = 1.0f;

        // 種族補正
        modifier *= Race.SanityModifier;

        // 宗教補正
        if (Religion != null)
        {
            modifier *= Religion.SanityModifier;
        }

        // スキル補正
        if (Skills.HasSkill("鋼の精神"))
        {
            modifier *= 0.9f;
        }

        return (int)(baseAmount * modifier);
    }

    public void SetSanity(int value) => Sanity = value;
    #endregion

    #region Hunger (満腹度) — Ver.prt.0.25改修: 10段階制
    private int _hunger = GameConstants.InitialHunger;  // 初期値70
    public int Hunger
    {
        get => _hunger;
        private set => _hunger = Math.Clamp(value, GameConstants.MinHunger, GameConstants.MaxHunger);
    }
    // 値域: -10 〜 150

    public HungerStage HungerStage => Hunger switch
    {
        >= 120 => HungerStage.Nausea,           // 吐き気
        >= 100 => HungerStage.Overeating,       // 過食
        >= 80  => HungerStage.Full,             // 満腹
        >= 50  => HungerStage.Normal,           // 通常
        >= 40  => HungerStage.SlightlyHungry,   // 空腹・小
        >= 0   => HungerStage.VeryHungry,       // 空腹・大
        >= -8  => HungerStage.Starving,         // 飢餓
        >= -9  => HungerStage.NearStarvation,   // 餓死寸前
        _      => HungerStage.Starvation        // 餓死（-10）
    };

    public event EventHandler<HungerStageChangedEventArgs>? OnHungerStageChanged;

    public void ModifyHunger(int amount)
    {
        var oldStage = HungerStage;
        Hunger += amount;
        if (HungerStage != oldStage)
            OnHungerStageChanged?.Invoke(this, new(oldStage, HungerStage));
    }
    #endregion

    #region Thirst (渇き度) — Ver.prt.0.25改修: 10段階制
    private int _thirst = GameConstants.InitialThirst;  // 初期値70
    public int Thirst
    {
        get => _thirst;
        private set => _thirst = Math.Clamp(value, GameConstants.MinThirst, GameConstants.MaxThirst);
    }

    public ThirstStage ThirstStage => Thirst switch
    {
        >= 120 => ThirstStage.Nausea,
        >= 100 => ThirstStage.Overdrinking,
        >= 80  => ThirstStage.Full,
        >= 50  => ThirstStage.Normal,
        >= 40  => ThirstStage.SlightlyThirsty,
        >= 0   => ThirstStage.VeryThirsty,
        >= -8  => ThirstStage.Dehydrated,
        >= -9  => ThirstStage.NearDesiccation,
        _      => ThirstStage.Desiccation
    };

    public event EventHandler<ThirstStageChangedEventArgs>? OnThirstStageChanged;

    public void ModifyThirst(int amount)
    {
        var oldStage = ThirstStage;
        Thirst += amount;
        if (ThirstStage != oldStage)
            OnThirstStageChanged?.Invoke(this, new(oldStage, ThirstStage));
    }
    #endregion

    #region Equipment & Inventory
    public Equipment Equipment { get; } = new();
    public Inventory Inventory { get; } = new(maxSlots: 30);

    protected override IEnumerable<StatModifier> GetAllStatModifiers()
    {
        return base.GetAllStatModifiers()
            .Concat(Equipment.GetStatModifiers());
    }

    protected override float GetResistanceAgainst(Element element)
    {
        return base.GetResistanceAgainst(element) + Equipment.GetElementalResistance(element);
    }
    #endregion

    #region Skills
    public SkillSet Skills { get; } = new();

    public bool LearnSkill(SkillData skill)
    {
        if (Skills.CanLearn(skill, this))
        {
            Skills.Add(skill);
            OnSkillLearned?.Invoke(this, new SkillEventArgs(skill));
            return true;
        }
        return false;
    }

    public SkillResult UseSkill(string skillId, GameState state, Character? target = null)
    {
        var skill = Skills.Get(skillId);
        if (skill == null) return SkillResult.NotFound;

        if (CurrentSp < skill.SpCost) return SkillResult.InsufficientSp;
        if (CurrentMp < skill.MpCost) return SkillResult.InsufficientMp;

        CurrentSp -= skill.SpCost;
        CurrentMp -= skill.MpCost;

        return skill.Execute(this, state, target);
    }
    #endregion

    #region Magic Language (魔法言語)
    public MagicLanguage MagicLanguage { get; } = new();

    public bool LearnMagicWord(MagicWordData word)
    {
        if (MagicLanguage.CanLearn(word))
        {
            MagicLanguage.Add(word);
            OnMagicWordLearned?.Invoke(this, new MagicWordEventArgs(word));
            return true;
        }
        return false;
    }

    public SpellResult CastSpell(string incantation, GameState state)
    {
        var parseResult = MagicLanguage.ParseIncantation(incantation);
        if (!parseResult.IsValid)
        {
            return SpellResult.InvalidIncantation(parseResult.Error);
        }

        var spell = parseResult.Spell;
        if (CurrentMp < spell.MpCost)
        {
            return SpellResult.InsufficientMp;
        }

        CurrentMp -= spell.MpCost;
        return spell.Cast(this, state);
    }
    #endregion

    #region Religion (宗教)
    public Religion? Religion { get; private set; }

    public void JoinReligion(Religion religion)
    {
        if (Religion != null)
        {
            LeaveReligion();
        }

        Religion = religion;
        OnReligionJoined?.Invoke(this, new ReligionEventArgs(religion));
    }

    public void LeaveReligion()
    {
        if (Religion == null) return;

        var oldReligion = Religion;
        Religion = null;

        // 背教ペナルティ
        ApplyStatusEffect(new StatusEffect(StatusEffectType.Apostasy, duration: 10000));

        OnReligionLeft?.Invoke(this, new ReligionEventArgs(oldReligion));
    }

    public void Pray()
    {
        Religion?.OnPray(this);
    }

    public void Sacrifice(Item item)
    {
        if (Religion?.AcceptSacrifice(item, this) == true)
        {
            Inventory.Remove(item);
        }
    }
    #endregion

    #region Knowledge (引き継ぎ対象知識)
    public PlayerKnowledge Knowledge { get; } = new();
    #endregion

    #region Flags (フラグ)
    public PlayerFlags Flags { get; } = new();
    #endregion

    #region Death & Rebirth
    public void OnDeath(DeathCause cause)
    {
        // 死亡イベント発火
        OnPlayerDeath?.Invoke(this, new PlayerDeathEventArgs(cause));
    }

    public TransferData ExtractTransferData(bool isSanityZero)
    {
        if (isSanityZero)
        {
            // 正気度0: フラグと実績のみ保持
            return new TransferData
            {
                Flags = Flags.Clone(),
                Achievements = Knowledge.Achievements.Clone(),
                Skills = new SkillSet(),
                MagicLanguage = new MagicLanguage(),
                Religion = null,
                NpcKnowledge = new Dictionary<string, NpcKnowledge>(),
                MapKnowledge = new Dictionary<string, MapKnowledge>(),
                EventKnowledge = new Dictionary<string, EventKnowledge>(),
                Recipes = new HashSet<string>()
            };
        }
        else
        {
            // 通常の死に戻り: 全知識を引き継ぎ
            return new TransferData
            {
                Flags = Flags.Clone(),
                Achievements = Knowledge.Achievements.Clone(),
                Skills = Skills.Clone(),
                MagicLanguage = MagicLanguage.Clone(),
                Religion = Religion?.CloneWithReducedFaith(0.8f),
                NpcKnowledge = Knowledge.Npcs.ToDictionary(k => k.Key, v => v.Value.Clone()),
                MapKnowledge = Knowledge.Maps.ToDictionary(k => k.Key, v => v.Value.Clone()),
                EventKnowledge = Knowledge.Events.ToDictionary(k => k.Key, v => v.Value.Clone()),
                Recipes = new HashSet<string>(Knowledge.Recipes)
            };
        }
    }

    public void ApplyTransferData(TransferData data)
    {
        // フラグ適用
        Flags.Apply(data.Flags);

        // 知識系適用（正気度0の場合は空）
        foreach (var skill in data.Skills.GetAll())
        {
            Skills.Add(skill);
        }

        foreach (var word in data.MagicLanguage.GetAll())
        {
            MagicLanguage.Add(word);
        }

        if (data.Religion != null)
        {
            JoinReligion(data.Religion);
        }

        foreach (var (key, value) in data.NpcKnowledge)
        {
            Knowledge.Npcs[key] = value;
        }

        foreach (var (key, value) in data.MapKnowledge)
        {
            Knowledge.Maps[key] = value;
        }

        foreach (var (key, value) in data.EventKnowledge)
        {
            Knowledge.Events[key] = value;
        }

        foreach (var recipe in data.Recipes)
        {
            Knowledge.Recipes.Add(recipe);
        }
    }
    #endregion

    #region Turn Actions
    public override TurnAction DecideAction(GameState state)
    {
        // プレイヤーは入力待ち
        return TurnAction.WaitForInput;
    }

    public override void ExecuteAction(TurnAction action, GameState state)
    {
        // アクションに応じた処理
        switch (action.Type)
        {
            case TurnActionType.Move:
                Move(action.Direction);
                break;
            case TurnActionType.Attack:
                Attack(action.Target, state);
                break;
            case TurnActionType.UseSkill:
                UseSkill(action.SkillId, state, action.Target);
                break;
            case TurnActionType.CastSpell:
                CastSpell(action.Incantation, state);
                break;
            case TurnActionType.UseItem:
                UseItem(action.ItemId, state);
                break;
            case TurnActionType.Wait:
                Wait();
                break;
        }

        // ターン経過処理
        DecreaseHunger(1);
        TickStatusEffects();
    }

    private void Move(Direction direction)
    {
        Position = Position.Move(direction);
        FacingDirection = direction;
    }

    private void Attack(Character target, GameState state)
    {
        // 基本攻撃処理
        var weapon = Equipment.MainHand;
        var attackType = weapon?.AttackType ?? AttackType.Unarmed;
        state.CombatSystem.ExecuteAttack(this, target, attackType);
    }

    private void Wait()
    {
        // 待機：SP回復
        RestoreSp(5);
    }
    #endregion

    #region Events
    public event EventHandler<LevelUpEventArgs>? OnLevelUp;
    public event EventHandler<SanityStageEventArgs>? OnSanityStageChanged;
```

```csharp
namespace RougelikeGame.Core.Entities;

/// <summary>
/// 敵キャラクター
/// </summary>
public class Enemy : Character
{
    #region Enemy Properties
    public string EnemyId { get; init; }
    public EnemyType EnemyType { get; init; }
    public int ThreatLevel { get; init; }
    public int ExperienceReward { get; init; }
    public LootTable LootTable { get; init; }
    #endregion

    #region AI
    public AIBehavior Behavior { get; set; }
    public AIState CurrentAIState { get; private set; } = AIState.Idle;
    public Character? CurrentTarget { get; private set; }

    private int _alertLevel;
    public int AlertLevel
    {
        get => _alertLevel;
        set => _alertLevel = Math.Clamp(value, 0, 100);
    }
    #endregion

    #region Detection
    public int DetectionRange => EffectiveStats.Perception;
    public int AttackRange { get; init; } = 1;

    public bool CanDetect(Character target, Map map)
    {
        int distance = Position.DistanceTo(target.Position);
        if (distance > DetectionRange) return false;

        // 視線チェック
        return map.HasLineOfSight(Position, target.Position);
    }
    #endregion

    #region Turn System
    public override TurnAction DecideAction(GameState state)
    {
        UpdateAIState(state);

        return CurrentAIState switch
        {
            AIState.Idle => DecideIdleAction(state),
            AIState.Patrol => DecidePatrolAction(state),
            AIState.Alert => DecideAlertAction(state),
            AIState.Combat => DecideCombatAction(state),
            AIState.Flee => DecideFleeAction(state),
            _ => TurnAction.Wait
        };
    }

    private void UpdateAIState(GameState state)
    {
        var player = state.Player;
        bool canSeePlayer = CanDetect(player, state.CurrentMap);
        int distanceToPlayer = Position.DistanceTo(player.Position);

        // HP低下時の逃走判定
        if (CurrentHp < MaxHp * 0.2f && Behavior.CanFlee)
        {
            CurrentAIState = AIState.Flee;
            return;
        }

        if (canSeePlayer)
        {
            CurrentTarget = player;
            AlertLevel = 100;
            CurrentAIState = AIState.Combat;
        }
        else if (AlertLevel > 0)
        {
            AlertLevel -= 5;
            CurrentAIState = AlertLevel > 50 ? AIState.Alert : AIState.Patrol;
        }
        else
        {
            CurrentAIState = AIState.Idle;
            CurrentTarget = null;
        }
    }

    private TurnAction DecideIdleAction(GameState state)
    {
        // ランダムに待機または移動
        if (state.Random.NextDouble() < 0.7)
        {
            return TurnAction.Wait;
        }

        var direction = (Direction)state.Random.Next(4);
        if (state.CurrentMap.CanMoveTo(Position.Move(direction)))
        {
            return TurnAction.Move(direction);
        }

        return TurnAction.Wait;
    }

    private TurnAction DecidePatrolAction(GameState state)
    {
        // パトロールルートに沿って移動
        if (Behavior.PatrolRoute != null && Behavior.PatrolRoute.Count > 0)
        {
            var nextPoint = Behavior.PatrolRoute[Behavior.CurrentPatrolIndex];
            var direction = GetDirectionTo(nextPoint);

            if (Position == nextPoint)
            {
                Behavior.CurrentPatrolIndex = (Behavior.CurrentPatrolIndex + 1) % Behavior.PatrolRoute.Count;
            }

            return TurnAction.Move(direction);
        }

        return DecideIdleAction(state);
    }

    private TurnAction DecideAlertAction(GameState state)
    {
        // 最後に見た位置に向かう
        if (Behavior.LastKnownTargetPosition.HasValue)
        {
            var direction = GetDirectionTo(Behavior.LastKnownTargetPosition.Value);
            return TurnAction.Move(direction);
        }

        return DecidePatrolAction(state);
    }

    private TurnAction DecideCombatAction(GameState state)
    {
        if (CurrentTarget == null) return TurnAction.Wait;

        int distance = Position.DistanceTo(CurrentTarget.Position);

        // 攻撃範囲内
        if (distance <= AttackRange)
        {
            // スキル使用判定
            var skill = SelectSkillToUse();
            if (skill != null && CurrentSp >= skill.SpCost)
            {
                return TurnAction.UseSkill(skill.Id, CurrentTarget);
            }

            return TurnAction.Attack(CurrentTarget);
        }

        // 接近
        var direction = GetDirectionTo(CurrentTarget.Position);
        if (state.CurrentMap.CanMoveTo(Position.Move(direction)))
        {
            return TurnAction.Move(direction);
        }

        // 迂回路を探す
        var alternatePath = state.Pathfinder.FindPath(Position, CurrentTarget.Position, state.CurrentMap);
        if (alternatePath != null && alternatePath.Count > 0)
        {
            return TurnAction.Move(GetDirectionTo(alternatePath[0]));
        }

        return TurnAction.Wait;
    }

    private TurnAction DecideFleeAction(GameState state)
    {
        if (CurrentTarget == null) return TurnAction.Wait;

        // ターゲットから離れる方向に逃走
        var fleeDirection = GetOppositeDirection(GetDirectionTo(CurrentTarget.Position));

        if (state.CurrentMap.CanMoveTo(Position.Move(fleeDirection)))
        {
            return TurnAction.Move(fleeDirection);
        }

        // 他の方向を試す
        foreach (Direction dir in Enum.GetValues<Direction>())
        {
            if (dir != GetDirectionTo(CurrentTarget.Position) && 
                state.CurrentMap.CanMoveTo(Position.Move(dir)))
            {
                return TurnAction.Move(dir);
            }
        }

        return TurnAction.Wait;
    }

    private SkillData? SelectSkillToUse()
    {
        // 使用可能なスキルからランダムに選択
        var availableSkills = Behavior.Skills
            .Where(s => CurrentSp >= s.SpCost && CurrentMp >= s.MpCost)
            .ToList();

        if (availableSkills.Count == 0) return null;

        // 優先度に基づいて選択（後で実装）
        return availableSkills[0];
    }

    private Direction GetDirectionTo(Position target)
    {
        int dx = target.X - Position.X;
        int dy = target.Y - Position.Y;

        if (Math.Abs(dx) > Math.Abs(dy))
        {
            return dx > 0 ? Direction.East : Direction.West;
        }
        else
        {
            return dy > 0 ? Direction.South : Direction.North;
        }
    }

    private Direction GetOppositeDirection(Direction dir) => dir switch
    {
        Direction.North => Direction.South,
        Direction.South => Direction.North,
        Direction.East => Direction.West,
        Direction.West => Direction.East,
        _ => Direction.North
    };

    public override void ExecuteAction(TurnAction action, GameState state)
    {
        switch (action.Type)
        {
            case TurnActionType.Move:
                Position = Position.Move(action.Direction);
                break;
            case TurnActionType.Attack:
                if (action.Target != null)
                {
                    state.CombatSystem.ExecuteAttack(this, action.Target, AttackType.Physical);
                }
                break;
            case TurnActionType.UseSkill:
                if (action.Target != null && Behavior.Skills.FirstOrDefault(s => s.Id == action.SkillId) is SkillData skill)
                {
                    CurrentSp -= skill.SpCost;
                    CurrentMp -= skill.MpCost;
                    skill.Execute(this, state, action.Target);
                }
                break;
            case TurnActionType.Wait:
                RestoreSp(2);
                break;
        }

        TickStatusEffects();
        Behavior.LastKnownTargetPosition = CurrentTarget?.Position;
    }
    #endregion

    #region Loot
    public List<Item> GenerateLoot(Random random)
    {
        return LootTable.Generate(random);
    }
    #endregion
}
```

```csharp
/// <summary>
/// 2次元座標を表す値オブジェクト
/// </summary>
public readonly record struct Position(int X, int Y)
{
    public static Position Zero => new(0, 0);

    public int DistanceTo(Position other) =>
        Math.Abs(X - other.X) + Math.Abs(Y - other.Y);

    public double EuclideanDistanceTo(Position other) =>
        Math.Sqrt(Math.Pow(X - other.X, 2) + Math.Pow(Y - other.Y, 2));

    public Position Move(Direction direction) => direction switch
    {
        Direction.North => this with { Y = Y - 1 },
        Direction.South => this with { Y = Y + 1 },
        Direction.East => this with { X = X + 1 },
        Direction.West => this with { X = X - 1 },
        Direction.NorthEast => new Position(X + 1, Y - 1),
        Direction.NorthWest => new Position(X - 1, Y - 1),
        Direction.SouthEast => new Position(X + 1, Y + 1),
        Direction.SouthWest => new Position(X - 1, Y + 1),
        _ => this
    };

    public IEnumerable<Position> GetNeighbors(bool includeDiagonals = false)
    {
        yield return Move(Direction.North);
        yield return Move(Direction.South);
        yield return Move(Direction.East);
        yield return Move(Direction.West);

        if (includeDiagonals)
        {
            yield return Move(Direction.NorthEast);
            yield return Move(Direction.NorthWest);
            yield return Move(Direction.SouthEast);
            yield return Move(Direction.SouthWest);
        }
    }

    public static Position operator +(Position a, Position b) =>
        new(a.X + b.X, a.Y + b.Y);

    public static Position operator -(Position a, Position b) =>
        new(a.X - b.X, a.Y - b.Y);

    public override string ToString() => $"({X}, {Y})";
}
```

```csharp
        modifier *= actor.GetStatusEffectTurnModifier();

        return Math.Max(1, (int)(BaseTurnCost * modifier));
```

### 6.3 TurnManager（ターン管理）

```csharp
namespace RougelikeGame.Engine.TurnSystem;

/// <summary>
/// ターン制御を管理するクラス
/// </summary>
public class TurnManager : ITurnManager
{
    private readonly PriorityQueue<(ITurnActor Actor, long NextTurn), long> _turnQueue;
    private long _currentTurn;
    private CombatState _combatState = CombatState.Normal;

    public bool IsPlayerTurn => CurrentActor is Player;
    public ITurnActor? CurrentActor { get; private set; }
    public long CurrentTurn => _currentTurn;
    public CombatState CombatState 
    { 
        get => _combatState;
        set
        {
            if (_combatState != value)
            {
                var old = _combatState;
                _combatState = value;
                OnCombatStateChanged?.Invoke(this, new CombatStateChangedEventArgs(old, value));
            }
        }
    }

    public TurnManager()
    {
        _turnQueue = new PriorityQueue<(ITurnActor, long), long>();
    }

    public void RegisterActor(ITurnActor actor)
    {
        _turnQueue.Enqueue((actor, _currentTurn), _currentTurn);
    }

    public void UnregisterActor(ITurnActor actor)
    {
        // 優先度キューからの削除（再構築が必要）
        var items = new List<(ITurnActor, long)>();
        while (_turnQueue.Count > 0)
        {
            var item = _turnQueue.Dequeue();
            if (item.Actor != actor)
            {
                items.Add(item);
            }
        }
        foreach (var item in items)
        {
            _turnQueue.Enqueue(item, item.Item2);
        }
    }

    public void ProcessTurn(GameState state)
    {
        if (_turnQueue.Count == 0) return;

        // 次に行動するアクターを取得
        var (actor, scheduledTurn) = _turnQueue.Dequeue();
        CurrentActor = actor;
        _currentTurn = scheduledTurn;

        // アクションを決定
        var action = actor.DecideAction(state);

        // 入力待ちの場合はプレイヤー入力を待つ
        if (action.BaseTurnCost == 0 && actor is Player)
        {
            // 入力待ち状態（外部で処理）
            _turnQueue.Enqueue((actor, _currentTurn), _currentTurn);
            return;
        }

        // アクションを実行
        actor.ExecuteAction(action, state);

        // 次の行動ターンを計算してキューに戻す
        int turnCost = action.CalculateFinalCost((Character)actor, state);
        long nextTurn = _currentTurn + turnCost;

        // 生存チェック
        if (actor is Character character && character.IsAlive)
        {
            _turnQueue.Enqueue((actor, nextTurn), nextTurn);
        }

        // イベント発火
        OnTurnProcessed?.Invoke(this, new TurnProcessedEventArgs(actor, action, turnCost));
    }

    /// <summary>
    /// ゲーム内時間を取得（秒単位）
    /// </summary>
    public TimeSpan GetGameTime() => TimeSpan.FromSeconds(_currentTurn);

    /// <summary>
    /// ゲーム内日数を取得
    /// </summary>
    public int GetGameDay() => (int)(_currentTurn / 86400) + 1;

    /// <summary>
    /// ゲーム内時刻を取得（0-23時）
    /// </summary>
    public int GetGameHour() => (int)((_currentTurn % 86400) / 3600);

    public event EventHandler<TurnProcessedEventArgs>? OnTurnProcessed;
    public event EventHandler<CombatStateChangedEventArgs>? OnCombatStateChanged;
}
```
